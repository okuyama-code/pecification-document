## おすすめgem
```ruby
gem 'activeadmin'              # 管理画面
gem 'active_model_serializers' # APIのフォーマット整形
gem 'config'                   # 環境ごとの定数管理
gem 'devise'                   # 認証
gem 'devise-i18n'              # devise日本語化
gem 'devise_token_auth'        # APIエンドポイント認証
gem 'discard'                  # 論理削除
gem 'image_processing'         # image variantsを使うために必要
gem 'kaminari'                 # ページネーション
gem 'rack-cors'                # Cross-Origin Resource Sharering
gem 'rails-i18n'               # 日本語化
gem 'ransack'                  # 検索機能
gem 'sassc-rails'              # コンパイルの高速化(activeadminに必要)
gem 'letter_opener_web'        # メール確認
gem 'brakeman'                 # 脆弱性検知
gem 'bullet'                   # クエリ検知
gem 'factory_bot_rails'        # テストデータ作成
gem 'rspec-rails'              # テストフレームワーク
gem 'rubocop'                  # 静的解析
gem 'pry-rails'                # デバッグ用
```
